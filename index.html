import yfinance as yf
import requests
import streamlit as st
from anthropic import Anthropic

client = Anthropic()  # uses ANTHROPIC_API_KEY env variable

def get_stock_data(ticker):
    stock = yf.Ticker(ticker)
    hist = stock.history(period="1mo")
    return hist["Close"].tolist(), hist["Close"].iloc[-1]

def get_headlines(ticker):
    # NewsAPI.org — free tier, 100 req/day
    url = f"https://newsapi.org/v2/everything?q={ticker}&apiKey=YOUR_KEY"
    r = requests.get(url).json()
    return [a["title"] for a in r["articles"][:5]]

def analyse_sentiment(headlines):
    # Claude API scores each headline
    prompt = f"""Analyse sentiment of these stock headlines.
Return JSON: {{"positive": 0-100, "neutral": 0-100, "negative": 0-100, "signal": "BUY/HOLD/SELL"}}
Headlines: {headlines}"""
    msg = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=200,
        messages=[{"role": "user", "content": prompt}]
    )
    return msg.content[0].text

# Streamlit UI
st.title("📈 AI Stock Sentiment Dashboard")
ticker = st.text_input("Enter ticker (e.g. BHP.AX or AAPL)")
if ticker:
    prices, current = get_stock_data(ticker)
    headlines = get_headlines(ticker)
    sentiment = analyse_sentiment(headlines)
    st.line_chart(prices)
    st.json(sentiment)
